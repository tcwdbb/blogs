``` js
    //防抖函数
		function debounce(fn, time) {
			let timer;
			return function(...args) {
				timer && clearTimeout(timer);
				timer = setTimeout(() => {
					fn.call(this, ...args);
				},time)
			}
		}
    //节流函数（利用时间戳实现）
		function throtile(fn, delay) {
			let prev = Date.now();
			return function(...args) {
				let now = Date.now();
				if(now - prev > delay) {
					fn.call(this, ...args)
					prev = Date.now();
				}
			}
		}
		//节流函数（利用延时器实现）
		function throtile1(fn, wait) {
			let timer;
			return (...args) => {
				if(timer) return;
				timer = setTimeout( () => {
					fn.call(...args);
					timer = null;
				},wait);
			}
		}
```

以下方法都会改变原数组
``` js
//删除并返回数组最后的一个元素
pop();
//在数组最后添加元素，可以添加多个，返回新的数组长度
push();
//删除数组第一个元素并返回这个元素
shift();
//在数组最前面添加元素，可以添加多个，返回新的数组长度
unshift();
//第一个参数是从下标为几开始删除，第二个参数是要删除几位，第三个参数是是要插入的元素，可以插入多个
splice();
//数组的排序，以字符编码进行排序
sort();
//从下到大排序
function sortNumber(a,b)
{
return a - b
}
arr.sort(sortNumber)
//颠倒数组中元素的顺序
reverse();
```
``` js
//将一个数组内元素以指定字符拼接成一个字符串
join();
```
